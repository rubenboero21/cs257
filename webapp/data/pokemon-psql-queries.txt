Select pokemon.name, generations.name
From pokemon, generations, linking_table
WHERE 1 = 1
AND pokemon.id = linking_table.pokemon_id
AND generations.id = linking_table.generation_id
ORDER by pokemon.name ASC;

*******all pokemon with grass type*******
SELECT pokemon.name, pokemon.dex_num
FROM pokemon, types, linking_table
WHERE 1 = 1
AND types.name = 'Grass'
AND pokemon.id = linking_table.pokemon_id
AND (types.id = linking_table.type1_id
OR types.id = linking_table.type2_id)
Order by pokemon.dex_num;

*******All pokemon with both grass and poison type*******
Select pokemon.name, pokemon.dex_num, t1.name, t2.name
FROM pokemon, types t1, types t2, linking_table
WHERE 1 = 1
AND t1.name = 'Grass'
AND t2.name = 'Poison'
AND pokemon.id = linking_table.pokemon_id
AND (t1.id = linking_table.type1_id
OR t1.id = linking_table.type2_id)
AND (t2.id = linking_table.type1_id
OR t2.id = linking_table.type2_id)
Order by pokemon.dex_num;

*****All pokemon with grass as primary and any secondary*****
Select pokemon.name, pokemon.dex_num, t1.name, t2.name
FROM pokemon, types t1, types t2, linking_table
WHERE 1 = 1
AND t1.name = 'Poison'
AND t2.name != ''
AND pokemon.id = linking_table.pokemon_id
AND t1.id = linking_table.type1_id
AND t2.id = linking_table.type2_id
Order by pokemon.dex_num;

*****All pokemon with 2 type, where 1 is poison type*****
Select pokemon.name, pokemon.dex_num, t1.name, t2.name
FROM pokemon, types t1, types t2, linking_table
WHERE 1 = 1
AND t1.name = 'Poison'
AND t2.name != ''
AND t2.name != t1.name
AND pokemon.id = linking_table.pokemon_id
AND (t1.id = linking_table.type1_id
OR t1.id = linking_table.type2_id)
AND (t2.id = linking_table.type1_id
OR t2.id = linking_table.type2_id)
Order by pokemon.dex_num;
